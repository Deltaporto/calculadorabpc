Fluxo de Cálculo Administrativo

flowchart TD
    subgraph Inputs
    Amb[Fatores Ambientais
(e1-e5)]
    Ativ[Atividades e Part.
(d1-d9)]
    Corpo[Funções do Corpo
(b1-b8)]
    end

    subgraph Calculations
    Amb -->|Soma x 5%| Q_Amb[Qualificador Amb
(0-4)]
    Ativ -->|Soma x 2.77%| Q_Ativ[Qualificador Ativ
(0-4)]
    Corpo -->|Máximo| Max_Corpo[Máximo Corpo]
    
    Max_Corpo --> CheckMaj{Majoração?}
    CheckMaj -- Sim --> Maj[+1 (max 4)]
    CheckMaj -- Não --> Q_Corpo[Qualificador Corpo
(0-4)]
    Maj --> Q_Corpo
    end

    subgraph TabelaConclusiva ["Tabela Conclusiva (Anexo IV)"]
    Q_Corpo --> CheckBlock{Corpo <= 1
OU
Ativ <= 1 ?}
    Q_Ativ --> CheckBlock
    
    CheckBlock -- Sim --> Indeferido[INDEFERIDO]
    CheckBlock -- Não --> CheckDefer{Corpo >= 3
OU
Ativ >= 3
OU
Amb >= 3 ?}
    
    CheckDefer -- Sim --> Deferido[DEFERIDO]
    CheckDefer -- Não --> Indeferido
    end

Fluxo de Controle Judicial

flowchart TD
    Start((Início)) --> BaseAdmin[1. Fixar Base Administrativa
(Amb, Ativ, Corpo)]
    BaseAdmin --> MedInfo[2. Informações Médicas Judiciais]
    
    MedInfo --> Impedimento{Impedimento
Longo Prazo?}
    Impedimento -- Não --> DispensaLP[DISPENSA Avaliação Social
(Suficiência da prova médica)]
    
    Impedimento -- Sim --> DefCorpo[Definir Corpo Judicial]
     
    subgraph CorpoJudLogic [Lógica Funções do Corpo]
    DefCorpo --> KeepCorp{Manter Admin?}
    KeepCorp -- Sim --> Q_Corpo_Jud[= Corpo Admin]
    KeepCorp -- Não --> ChangeReason{Motivo?}
    ChangeReason -- Estrutura/Prog --> Q_Corpo_Jud_Calc[Corpo Admin + 1]
    ChangeReason -- Domínio Max --> Q_Corpo_Jud_Max[Máximo b1-b8]
    ChangeReason -- Reclassificação --> Q_Corpo_Jud_Man[Manual]
    Q_Corpo_Jud_Calc --> Q_Corpo_Jud
    Q_Corpo_Jud_Max --> Q_Corpo_Jud
    Q_Corpo_Jud_Man --> Q_Corpo_Jud
    end
    
    Q_Corpo_Jud --> TesteA{Teste A:
Tabela Conclusiva
(Amb Admin + Ativ Admin + Corpo Jud)}
    
    TesteA -- DEFERIDO --> DispensaA[DISPENSA Avaliação Social
(Deferido por prova médica)]
    
    TesteA -- INDEFERIDO --> CheckIrrelevancia{Corpo Jud <= 1?}
    CheckIrrelevancia -- Sim --> DispensaB[DISPENSA Avaliação Social
(Impossível deferir independente de Ativ/Amb)]
    
    CheckIrrelevancia -- Não --> CheckAtivMed{Perito requalificou
Atividades?}
    
    CheckAtivMed -- Sim --> TesteB{Teste B:
Tabela Conclusiva
(Amb Admin + Ativ Méd + Corpo Jud)}
    TesteB -- DEFERIDO --> DispensaC[DISPENSA Avaliação Social
(Deferido com reclass. de Atividades)]
    TesteB -- INDEFERIDO --> SocialNec[AVALIAÇÃO SOCIAL
NECESSÁRIA]
    
    CheckAtivMed -- Não --> SocialNec

    class Start,BaseAdmin,MedInfo start

    class SocialNec end_social
